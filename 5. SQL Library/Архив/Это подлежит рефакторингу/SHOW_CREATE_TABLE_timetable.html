timetableCREATE TABLE `timetable` (
`ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
`date` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT 'Дата и время занятия',
`course` int(10) unsigned NOT NULL DEFAULT '0' COMMENT 'Название курса',
`teacher` int(10) unsigned NOT NULL DEFAULT '0' COMMENT 'Преподаватель, который будет вести занятие',
PRIMARY KEY (`ID`),
KEY `course_2` (`course`),
KEY `teacher_2` (`teacher`),
CONSTRAINT `fkCourse` FOREIGN KEY (`course`) REFERENCES `courses` (`ID`) ON UPDATE CASCADE,
CONSTRAINT `fkTeacher` FOREIGN KEY (`teacher`) REFERENCES `teachers` (`ID`) ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=13 DEFAULT CHARSET=utf8
